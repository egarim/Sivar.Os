━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  This is a well-known ASP.NET Core gotcha:
  
  By default, JwtSecurityTokenHandler transforms OIDC claims to WS-Fed schema
  for backwards compatibility. Setting DefaultMapInboundClaims = false 
  preserves the standard OIDC claim names.
  
  This MUST be set BEFORE AddAuthentication() - order matters!
